CREATE TABLE HABILITACAO
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	NOME VARCHAR(2);
);

CREATE TABLE USUARIO
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	LOGIN VARCHAR(20) NOT NULL UNIQUE,
	SENHA VARCHAR(32),
	TIPO INTEGER
);

CREATE TABLE VEICULO
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	MARCA VARCHAR(20),
	MODELO VARCHAR(20),
	PLACA VARCHAR(8) UNIQUE,
	ID_HABILITACAO INTEGER FOREIGN KEY REFERENCES HABILITACAO(ID)
);

CREATE TABLE MOTORISTA
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	NOME VARCHAR(20),
	ID_HABILITACAO INTEGER FOREIGN KEY REFERENCES HABILITACAO(ID)
);

CREATE TABLE CORRIDA
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	DESTINO VARCHAR(50),
	DATASAIDA DATETIME NOT NULL,
	PREVISAOCHEGADA DATETIME NOT NULL,
	DATACHEGADA DATETIME,
	QUILOMETRAGEM DOUBLE,
	ID_VEICULO INTEGER FOREIGN KEY REFERENCES VEICULO(ID)
);

CREATE TABLE MOTORISTAFAZCORRIDA
(
	ID_CORRIDA INTEGER FOREIGN KEY REFERENCES CORRIDA(ID),
	ID_MOTORISTA INTEGER FOREIGN KEY REFERENCES MOTORISTA(ID)
);

CREATE TABLE CATEGORIA
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	NOME VARCHAR(50) NOT NULL UNIQUE,
	DESCRICAO VARCHAR(200)
);

CREATE TABLE MANOBRA
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	IDUSUARIO INTEGER FOREIGN KEY REFERENCES USUARIO(ID),
	NOME VARCHAR(50),
	DIFICULDADE INTEGER,
	STATUS INTEGER,
	ID_CATEGORIA INTEGER FOREIGN KEY REFERENCES CATEGORIA(ID)
);


CREATE TABLE TREINO
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	IDMANOBRA INTEGER FOREIGN KEY REFERENCES MANOBRA(ID),
	PROGRESSO INTEGER,
	QNTDDIAS INTEGER,
	IDUSUARIO INTEGER FOREIGN KEY REFERENCES USUARIO(ID)
);

CREATE TABLE REQUISITO
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	IDMANOBRA INTEGER,
	IDMANOBRAREQUISITO INTEGER FOREIGN KEY REFERENCES MANOBRA(ID)
);

CREATE TABLE DICA
(
	ID INTEGER NOT NULL PRIMARY KEY IDENTITY,
	IDMANOBRA INTEGER FOREIGN KEY REFERENCES MANOBRA(ID),
	DESCRICAO VARCHAR(200),
	IDUSUARIO INTEGER FOREIGN KEY REFERENCES USUARIO(ID)
);
